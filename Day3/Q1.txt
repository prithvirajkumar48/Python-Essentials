# Program to check Armstrong numbers in a certain interval
lower = 1042000
upper = 702648265 
count = 0
print("The First Armstrong Number between ", lower, "and", upper, "is :")

for num in range(lower, upper + 1):

   # order of number
   order = len(str(num))
    
   # initialize sum
   sum = 0

   temp = num
   while temp > 0:
       digit = temp % 10
       sum += digit ** order
       temp //= 10

   if num == sum:
       count = count + 1
       if count==1:
           print(num)
       else:
           exit()
